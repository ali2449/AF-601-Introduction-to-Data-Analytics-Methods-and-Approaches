Sample Solution
CREATE TABLE customer_feedback ( feedback_id INTEGER PRIMARY KEY, customer_id INTEGER NOT NULL, feedback_date DATE DEFAULT CURRENT_DATE, rating INTEGER CHECK (rating BETWEEN 1 AND 5), comments TEXT );
